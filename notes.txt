# Moth and Money V4 — run locally (macOS / zsh)

# From the project folder (the one that contains the `app/` directory):

# Activate the virtual environment (run this in every new terminal session):
source app/.venv/bin/activate

# Run the Streamlit app:
streamlit run app/app.py

# By default the app uses bundled demo data (USE_SAMPLE_DATA=true). No database is required.

# ---
# First-time Python setup (only once), if you do not have a venv yet:
#   cd app
#   python3 -m venv .venv
#   source .venv/bin/activate
#   pip install -r requirements.txt
#   cd ..

# ---
# PostgreSQL (optional — live data from the database)

# 1. Create a database (example using psql):
#    createdb moth_and_money
#    psql -d moth_and_money -f app/db/schema.sql
#    psql -d moth_and_money -f app/db/seed_demo.sql

# 2. Copy app/.env.example to app/.env and set DB_HOST, DB_PORT, DB_NAME, DB_USER, DB_PASSWORD
#    (or set DATABASE_URL). Set USE_SAMPLE_DATA=false so pages read from PostgreSQL.

# 3. Run the app as above. If the database is unreachable, the app will show an error until
#    USE_SAMPLE_DATA=true or the connection is fixed.

# If the schema was applied before a new column was added (e.g. artist_title), run:
#    psql -d moth_and_money -f app/db/migrations/001_add_artist_title.sql

# ---
# Reset PostgreSQL for a clean test run (destroys all data in DB_NAME)

# From repo root with app/.venv activated and app/.env configured for Postgres:
#   python scripts/reset_database.py              # prompts: type YES to confirm
#   python scripts/reset_database.py --yes        # no prompt
#   python scripts/reset_database.py --yes --with-seed   # also load seed_demo.sql

# Requires psql on your PATH. Stop Streamlit first if it holds DB connections.
